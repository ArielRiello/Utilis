# Cria ambiente virtual:
python -m venv venv

# Ativa ambiente virtual:
.\venv\Scripts\Activate

# Executa somente scripts criados no computador (não executa baixados da internet):
Set-ExecutionPolicy -ExecutionPolicy RemoteSigned -Scope CurrentUser

# Libera execução de scripts temporariamente:
Set-ExecutionPolicy -ExecutionPolicy Bypass -Scope Process

# Desativa ambiente virtual:
deactivate

# Gera o arquivo "requirements.txt":
pip freeze > requirements.txt

# Atualiza todas bibliotecas do "requirements.txt":
pip install --upgrade -r requirements.txt

# Atualiza o "requirements.txt":
pip freeze > requirements.txt

# Instala todas bibliotecas do "requirements.txt":
pip install -r requirements.txt

# Criar ambiente virtual especificando o caminho completo para o Python (exemplo genérico):
# Você pode escolher uma versão específica do Python.
C:\Caminho\Para\Python\python.exe -m venv venv

# Verificar a versão do Python no ambiente virtual:
python --version
